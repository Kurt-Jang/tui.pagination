        itemCount: 500,
        itemPerPage : 20,
        pagePerPageList: 5,
        moveUnit: 'pagelist',
        isCenterAlign: true,
        insertTextNode: '',
        classPrefix: '',
        firstItemClassName: 'first-child',
        lastItemClassName: 'last-child',
        pageTemplate: '<a href="#">{=page}</a>',
        currentPageTemplate: '<strong>{=page}</strong>'
    }, $('.paginate1'));
