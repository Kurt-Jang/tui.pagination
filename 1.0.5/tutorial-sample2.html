    CustomEvent Example

        var pagination = new tui.component.Pagination({
        itemCount: 500
    }, $('.paginate1'));

    pagination.on('beforeMove', function(eventData) {
        $('#state').val(eventData.page + '로 이동합니다.\n' + $('#state').val());
    });

    pagination.on('afterMove', function(eventData) {
        $('#state').val(eventData.page + '로 이동되었습니다.\n' + $('#state').val());
    });
